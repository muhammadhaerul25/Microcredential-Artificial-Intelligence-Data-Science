def detect_outliers(df,x):
    Q1 = df[x].quantile(q=0.25)
    Q3 = df[x].quantile(q=0.75)
    IQR = Q3-Q1
    return df[(df[x] < Q1-1.5*IQR) | (df[x] > Q3+1.5*IQR)]
